# طريقة الاستخدام:

للبدء

g!start [room] [time] [winners] [gift]

g!start [منشن الروم] [الوقت] [عدد الفائزين] [الهدية]

للانهاء

g!end [giveaway id]

g!end [اي دي رسالة القيف اوي]

مثال:

g!start #info 15m 1 Nitro Gaming.

g!end 807930499125346315


امثل عن الوقت:

1d = 1 day

1h = 1 hour

1m = 1 minute

1s = 1 second
